namespace = verne_liliac

# Liliac War recount event - triggers when the recount mission completes
country_event = {
	id = verne_liliac.1
	title = verne_liliac.1.t
	desc = verne_liliac.1.d
	picture = BAD_WITH_MONARCH_eventPicture

	is_triggered_only = yes

	immediate = {
		set_country_flag = verne_liliac_war_recounted
		set_country_flag = verne_liliac_diplomacy_path_open
	}

	option = {
		name = verne_liliac.1.a
		add_stability = 1
		add_prestige = -10
		add_country_modifier = {
			name = verne_liliac_war_scarred
			duration = 3650
		}
	}

	option = {
		name = verne_liliac.1.b
		add_stability = -1
		add_country_modifier = {
			name = verne_liliac_war_vengeful
			duration = 3650
		}
	}
}

# Liliac unity through diplomacy event - late-game
country_event = {
	id = verne_liliac.4
	title = verne_liliac.4.t
	desc = verne_liliac.4.d
	picture = GOOD_WITH_MONARCH_eventPicture

	is_triggered_only = yes

	trigger = {
		tag = A33
		OR = {
			has_country_flag = verne_liliac_diplomacy_path_open
			has_country_flag = verne_liliac_vengeful_path
		}
	}

	option = {
		name = verne_liliac.4.a
		add_prestige = 10
		add_legitimacy = 10
		add_country_modifier = {
			name = verne_liliac_unity_through_diplomacy
			duration = 7300
		}
		if = {
			limit = { has_country_modifier = verne_liliac_war_scarred }
			remove_country_modifier = verne_liliac_war_scarred
		}
		if = {
			limit = { has_country_modifier = verne_liliac_war_vengeful }
			remove_country_modifier = verne_liliac_war_vengeful
		}
	}
}
