# ===============================================
# Verne overhaul dynasty safeguard chain (canonical)
# ===============================================

country_event = {
	id = verne_overhaul_dynasty.1
	title = verne_overhaul_dynasty.1.t
	desc = verne_overhaul_dynasty.1.d
	picture = GOOD_WITH_MONARCH_eventPicture

	is_triggered_only = yes
	hidden = yes

	trigger = {
		verne_overhaul_should_run_heir_safeguard_trigger = yes
	}

	immediate = {
		# Intent: always repair empty/placeholder heirs; optionally repair off-lineage heirs.
		if = {
			limit = { verne_overhaul_heir_requires_forced_correction_trigger = yes }
			country_event = { id = verne_overhaul_dynasty.2 }
		}
		else_if = {
			limit = {
				verne_overhaul_wrong_dynasty_heir_trigger = yes
				has_country_modifier = verne_dynasty_exalted_lineage
			}
			# Intent: exalted lineage is a hard lock on succession drift.
			country_event = { id = verne_overhaul_dynasty.2 }
		}
		else_if = {
			limit = {
				verne_overhaul_wrong_dynasty_heir_trigger = yes
				has_country_modifier = verne_dynasty_protected_court
			}
			# Intent: protected courts strongly prefer correction, but still permit occasional drift.
			random_list = {
				70 = { country_event = { id = verne_overhaul_dynasty.2 } }
				30 = { }
			}
		}
		else_if = {
			limit = { verne_overhaul_wrong_dynasty_heir_trigger = yes }
			# Intent: baseline safeguard catches only a minority of drift when no protection regime is active.
			random_list = {
				25 = { country_event = { id = verne_overhaul_dynasty.2 } }
				75 = { }
			}
		}
	}

	option = {
		name = verne_overhaul_dynasty.1.a
	}
}
