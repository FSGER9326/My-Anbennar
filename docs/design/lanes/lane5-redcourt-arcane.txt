# A33_fifth_sloth wrapper (existing — do not duplicate, this block goes inside it)

	# Commentary:
	# First Red Court mission. Establishes the mage advisory pipeline by proving
	# Verne can support a university and has embraced the Renaissance. The tier 3
	# reform verne_red_court_arcana_reform is the institutional gate — the player
	# must commit to the Red Court reform path early.
	# Lore: The Red Court is Verne's council of arcane advisors, formally woven
	# into the silver oath infrastructure. This mission formalizes that relationship.
	A33_establish_red_court = {
		icon = mission_magic
		position = 2
		required_missions = { A33_break_the_queen_of_the_hill }
		provinces_to_highlight = {
			owned_by = ROOT
			has_building = university
		}

		trigger = {
			has_institution = renaissance
			has_reform = verne_red_court_arcana_reform
			any_owned_province = {
				has_building = university
			}
		}

		effect = {
			add_advisor_pool = 1
			add_country_modifier = {
				name = "verne_red_court_advisory"
				duration = -1
			}
			custom_tooltip = verne_L5_establish_red_court_tt
			hidden_effect = {
				set_country_flag = verne_red_court_mage_advisors_enabled
			}
			country_event = { id = verne_overhaul_advisor.100 }
		}
	}

	# Commentary:
	# Mid-tier Red Court mission. Fuses military professionalism with arcane doctrine.
	# Requires the Dragonwake Ordinance reform — a deliberate power spike that
	# justifies fielding war wizards in conventional armies.
	# Lore: The Dragonwake Ordinance codifies the use of battle magic in
	# Vernman military formations, blending academy theory with battlefield doctrine.
	A33_dragonwake_ordinance = {
		icon = mission_wizard
		position = 5
		required_missions = { A33_establish_red_court }
		provinces_to_highlight = { }

		trigger = {
			has_reform = verne_dragonwake_ordinance_reform
		}

		effect = {
			add_country_modifier = {
				name = "verne_dragonwake_doctrine"
				duration = 7300
			}
			add_yearly_army_professionalism = 0.5
			add_country_modifier = {
				name = "verne_arcane_shock_pip"
				duration = 7300
			}
			define_general = {
				name = "Verne War Wizard"
				fire = 4
				shock = 6
				manuever = 3
				siege = 2
			}
			add_country_modifier = {
				name = "verne_temporary_war_wizard"
				duration = 7300	# 20 years
			}
			custom_tooltip = verne_L5_dragonwake_ordinance_tt
		}
	}

	# Commentary:
	# Late Red Court capstone. The Battle Mage Collegium represents full integration
	# of arcane warfare into the Vernman military state. This is the final reform
	# gate in the Red Court chain and unlocks a permanent military buff plus a
	# strategic event chain decision.
	# Lore: The Battle Mage Collegium is an elite institution training mages for
	# frontline combat roles — artillery support, fire bombardment, and morale
	# enchantment. It transforms Verne from a state that uses mages into a state
	# built around mage warfare.
	A33_battle_mage_collegium = {
		icon = mission_cannon
		position = 9
		required_missions = { A33_dragonwake_ordinance }
		provinces_to_highlight = {
			province_id = 288
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { has_building = university }
			}
		}

		trigger = {
			has_reform = verne_battle_mage_collegium_reform
			288 = {
				owned_by = ROOT
				has_building = university
			}
		}

		effect = {
			add_country_modifier = {
				name = "verne_battle_mage_artillery_fire"
				duration = -1
			}
			add_country_modifier = {
				name = "verne_battle_mage_land_fire"
				duration = -1
			}
			add_country_modifier = {
				name = "verne_battle_mage_morale"
				duration = -1
			}
			custom_tooltip = verne_L5_battle_mage_collegium_tt
			hidden_effect = {
				set_country_flag = verne_battle_mage_collegium_unlocked
			}
			country_event = { id = verne.200 }
		}
	}
