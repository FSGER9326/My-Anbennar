# Missions are ordered by position within A33_first_slot.
# Cross-lane dependencies use flags/variables, not required_missions from other slots.

	# Commentary (plain-language): Establishes Verne's early court identity by formalizing the Silver Oaths as permanent court doctrine.
	# Base vs new: base-patched (reform exists, mission is new)
	# Playstyle effect: Gives Verne a unique early-game court modifier reinforcing the reform-progress playstyle and advisor meta.
	# Risk note: modifier key verne_silver_oaths_formalized and flag silver_oaths_formalized must remain stable across patches.
	# Province context: no provinces — this is a government reform / court identity mission.
	# Lore-fit note: inferred-from-repo; the Court of Silver Oaths is Verne's foundational governing institution.
	A33_formalize_silver_oaths = {
		icon = imperial_diplomacy
		position = 1
		required_missions = { }
		provinces_to_highlight = { }

		trigger = {
			has_reform = verne_court_of_silver_oaths_reform
			NOT = { has_country_flag = silver_oaths_formalized }
		}

		effect = {
			set_country_flag = silver_oaths_formalized
			custom_tooltip = verne_A1L1_tt_formalize_silver_oaths
			add_country_modifier = {
				name = verne_silver_oaths_formalized
				duration = -1
			}
		}
	}

	# Commentary (plain-language): Opens the Marriage Court system, tying into dynasty safeguard events and noble house binding.
	# Base vs new: new (marriage court decision integration)
	# Playstyle effect: Encourages dynastic play and estate management; nobles become a key loyalty vector for court missions.
	# Risk note: depends on dynasty safeguard event system (verne_overhaul_dynasty_machine_started trigger). Keep flag verne_marriage_court_active stable.
	# Province context: no provinces — decision/court event driven.
	# Lore-fit note: inferred-from-repo; the Marriage Court is a Silver Oaths sub-institution for dynastic diplomacy.
	A33_open_marriage_court = {
		icon = royal_marriage
		position = 2
		required_missions = { A33_formalize_silver_oaths }
		provinces_to_highlight = { }

		trigger = {
			has_country_flag = silver_oaths_formalized
			NOT = { has_country_flag = verne_marriage_court_event_active }
		}

		effect = {
			set_country_flag = verne_marriage_court_active
			custom_tooltip = verne_A1L1_tt_open_marriage_court
			add_country_modifier = {
				name = verne_marriage_court_opened
				duration = -1
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 10
			}
			add_prestige = 15
		}
	}

	# Commentary (plain-language): Formalizes the great noble houses as Exalted Lineages, boosting heir quality and estate cohesion.
	# Base vs new: new (heir quality + dynasty safeguard integration)
	# Playstyle effect: Provides a mid-game heir-stat boost and deepens the nobles estate loop. Synergizes with dynasty safeguard events.
	# Risk note: define_heir approach must be tested for interaction with dynasty safeguard scripted effects. Keep modifier key stable.
	# Province context: no provinces — heir/estate/flag driven.
	# Lore-fit note: inferred-from-repo; the Exalted Lineages are the tier-2 noble institution under the Silver Oaths hierarchy.
	A33_exalted_lineages = {
		icon = strengthen_aristocracy
		position = 4
		required_missions = { A33_open_marriage_court }
		provinces_to_highlight = { }

		trigger = {
			has_reform = verne_charter_of_great_captains_reform
			has_country_flag = verne_marriage_court_active
		}

		effect = {
			custom_tooltip = verne_A1L1_tt_exalted_lineages
			# Heir quality boost: +1 average to all stats via define_heir with enhanced=True
			# This interacts with the dynasty safeguard system — if the safeguard fires,
			# the new heir also benefits from this modifier.
			define_heir = {
				adm = 3
				dip = 3
				mil = 3
				age = 15
				claim = 90
				dynasty = ROOT
			}
			add_country_modifier = {
				name = verne_exalted_lineages_legacy
				duration = -1
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 10
			}
			add_prestige = 10
			# Dynasty safeguard integration: set flag so the safeguard system knows
			# the Exalted Lineages are active and can apply enhanced heir generation.
			set_country_flag = verne_exalted_lineages_active
		}
	}

	# Commentary (plain-language): The capstone of the Court & Oaths lane — Verne's court becomes a world institution, attracting global talent.
	# Base vs new: new (world court capstone)
	# Playstyle effect: Diplomatic power spike in the late game. Synergizes with Vernissage, maritime, and political lanes via improve_relations and innovativeness.
	# Risk note: modifier verne_court_hegemon stacks with tier 7 reform bonuses — intentional but monitor for overpowered diplomatic output.
	# Province context: no provinces — global court modifier.
	# Lore-fit note: inferred-from-repo; the Vernman Court of the World is Verne's endgame diplomatic institution.
	A33_world_court_ascension = {
		icon = the_vienna_congress
		position = 8
		required_missions = { A33_exalted_lineages }
		provinces_to_highlight = { }

		trigger = {
			has_reform = verne_vernman_court_of_the_world_reform
			# Mid mission must be completed (exalted lineages flag serves as proxy)
			has_country_flag = verne_exalted_lineages_active
		}

		effect = {
			custom_tooltip = verne_A1L1_tt_world_court_ascension
			add_country_modifier = {
				name = verne_court_hegemon
				duration = -1
			}
			add_dip_power = 100
			change_innovativeness = 3
		}
	}
