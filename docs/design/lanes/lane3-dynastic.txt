# Verne Overhaul - Lane 3: Dynastic Machine
# File: docs/design/lanes/lane3-dynastic.txt
# Status: Design draft — Lane 3 (slot 3) of the 10-lane mission tree
# Last updated: 2026-03-31
#
# Theme: Dynastic legitimacy, noble privilege integration, and the
# mechanical reinforcement of the síl Verne bloodline. This lane is
# about making your dynasty an unstoppable engine — installing heirs,
# binding nobles, and culminating in a unique government reform that
# cements the Verne throne forever.
#
# Cross-lane deps:
# - Mission 1 (enact_dynastic_safeguard) gates on A33_formalize_silver_oaths
#   from Lane 1. This ensures the player has stabilised the court before
#   attempting dynastic engineering.
# - Mission 3 (exalted_dynasty_machine) gates on tier 8 reform
#   verne_throne_of_the_wyvern_kings_reform from Lane 1's reform ladder.
#
# Events referenced:
# - verne_overhaul_dynasty.1 — hidden event that installs a safeguarded heir
#   (defined in events/verne_overhaul_dynasty_events.txt)

A33_third_slot = {
	slot = 3
	generic = no
	ai = yes
	has_country_shield = yes
	potential = { tag = A33 }

	# ------------------------------------------------------------------
	# Mission 1: Enact Dynastic Safeguard (Early, position 1)
	# ------------------------------------------------------------------
	# Commentary:
	#   The opening mission of the dynastic lane. Triggers when Verne is
	#   in a vulnerable heir state (no heir or low legitimacy) AND has
	#   already formalised the Silver Oaths. Fires the dynasty safeguard
	#   event to install a competent heir and patches legitimacy.
	#   Playstyle effect: rewards players who rush Lane 1 reforms before
	#   their dynasty line runs dry. Acts as a safety net + minor buff.
	#   Design note: The hidden event verne_overhaul_dynasty.1 already
	#   checks its own heir-state trigger, so we rely on the mission
	#   trigger to gate access and let the event handle the install logic.
	A33_enact_dynastic_safeguard = {
		icon = mission_monarch_in_throne
		position = 1
		provinces_to_highlight = {
		}

		trigger = {
			OR = {
				NOT = { has_heir = yes }
				legitimacy = 50
				NOT = { legitimacy = 70 }
			}
			has_completed_mission = A33_formalize_silver_oaths
			verne_overhaul_is_verne_trigger = yes
		}

		effect = {
			# Fire the safeguard event to install heir + apply dynasty protection
			country_event = { id = verne_overhaul_dynasty.1 }
			add_legitimacy = 10
			add_country_modifier = {
				name = "verne_dynasty_safeguard_active"
				duration = -1
			}
			set_country_flag = verne_dynasty_protected
			custom_tooltip = verne_dynasty_safeguard_tt
		}
	}

	# ------------------------------------------------------------------
	# Mission 2: Expand Noble Privileges (Mid, position 4)
	# ------------------------------------------------------------------
	# Commentary:
	#   The mid-lane mission that deepens the bond between the Verne
	#   dynasty and the nobility. Requires the safeguard to be in place
	#   and the nobles estate to not be dangerously overpowered.
	#   Grants nobles loyalty, shaves crown land, and advances the
	#   dynasty projection variable — the abstracted score that tracks
	#   how far the player has pushed the dynastic machine.
	#   Playstyle effect: trade crown authority for noble goodwill and
	#   dynasty projection. Opens the door to the capstone reform.
	#   Risk note: crown land penalty is modest (-5%) to avoid punishing
	#   players who already leaned into the nobility estate.
	A33_expand_noble_privileges = {
		icon = mission_alliances
		position = 4
		provinces_to_highlight = {
		}

		required_missions = { A33_enact_dynastic_safeguard }

		trigger = {
			NOT = {
				estate_influence = {
					estate = estate_nobles
					influence = 80
				}
			}
			OR = {
				has_estate_privilege = estate_nobles_ducal_muster
				has_estate_privilege = estate_nobles_levies
				has_estate_privilege = estate_nobles_officer_corp
			}
			verne_overhaul_is_verne_trigger = yes
		}

		effect = {
			add_estate_nobles_loyalty_effect = yes
			change_estate_land_share = {
				estate = estate_nobles
				share = -5
			}
			change_variable = {
				which = verne_dynastic_magic_machine
				value = 1
			}
			custom_tooltip = verne_dynasty_projection_tt
			add_country_modifier = {
				name = "verne_noble_charter_renewed"
				duration = 9125
			}
		}
	}

	# ------------------------------------------------------------------
	# Mission 3: Exalted Dynasty Machine (Late, position 8)
	# ------------------------------------------------------------------
	# Commentary:
	#   The capstone of the dynastic lane. Gates on the tier 8 reform
	#   (verne_throne_of_the_wyvern_kings_reform) which is the final
	#   rung of the government reform ladder. Unlocks a unique
	#   government reform that turns the Verne dynasty into a self-
	#   sustaining heir-generation machine with unrest reduction.
	#   Integrates with exalted lineage decisions — the reward reform
	#   stacks with existing dynasty modifiers to make the bloodline
	#   effectively unbreakable.
	#   Playstyle effect: transforms the dynasty from a passive bonus
	#   into an active mechanical engine. The heir chance increase
	#   ensures the line never runs dry, and the unrest reduction
	#   reflects the stabilising effect of an undisputed succession.
	#   Lore note: "exalted dynasty machine" is the in-lore term for
	#   the Verne court's self-reinforcing legitimacy apparatus —
	#   a combination of marriage politics, noble compacts, and
	#   institutional safeguards.
	A33_exalted_dynasty_machine = {
		icon = mission_empire
		position = 8
		provinces_to_highlight = {
		}

		required_missions = { A33_expand_noble_privileges }

		trigger = {
			has_reform = verne_throne_of_the_wyvern_kings_reform
			verne_overhaul_is_verne_trigger = yes
			legitimacy = 90
			prestige = 50
		}

		effect = {
			# Swap to the unique exalted dynasty machine reform
			add_government_reform = verne_exalted_dynasty_machine_reform
			custom_tooltip = verne_exalted_dynasty_machine_tt
			set_country_flag = verne_exalted_dynasty_machine_unlocked
			add_prestige = 20
			add_legitimacy = 10
			add_country_modifier = {
				name = "verne_exalted_lineage_cemented"
				duration = -1
			}
		}
	}
}

# ======================================================================
# --- New modifiers ---
# ======================================================================

# verne_dynasty_safeguard_active
# Applied by: A33_enact_dynastic_safeguard
# Duration: permanent (flag-gated, removable by event decision)
# Effect: +1 yearly legitimacy, +10% heir chance, -1 national unrest
# Purpose: Signals that the dynasty safeguard is in effect. Makes the
#   heir-install event fire more reliably and gives passive legitimacy
#   recovery while the safeguard holds.

# verne_noble_charter_renewed
# Applied by: A33_expand_noble_privileges
# Duration: 25 years (9125 days)
# Effect: +10% nobles loyalty, +1 diplomatic reputation, +5% manpower
# Purpose: Temporary reward for deepening noble privileges. Provides a
#   mid-game boost without permanently shifting estate balance.

# verne_exalted_lineage_cemented
# Applied by: A33_exalted_dynasty_machine
# Duration: permanent
# Effect: +2 yearly legitimacy, +10% heir chance (stacks with safeguard),
#   -2 national unrest, +0.5 yearly prestige
# Purpose: The capstone dynasty modifier. Stacks with the safeguard to
#   make the Verne bloodline effectively unbreakable. Reflects the
#   culmination of the dynastic machine's self-reinforcing legitimacy.
