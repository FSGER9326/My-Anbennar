# Verne Overhaul - reusable scripted triggers
# Purpose: provide human-readable checks for common Verne progression states.

verne_overhaul_is_verne_or_subject = {
	OR = {
		is_or_was_tag = { tag = A33 }
		overlord = { is_or_was_tag = { tag = A33 } }
	}
}

verne_overhaul_has_adventure_network_started = {
	OR = {
		has_country_flag = verne_unlock_port_of_adventure_button
		has_country_modifier = verne_network_of_adventure_tier_1
	}
}

verne_overhaul_laments_regatta_anchor_state_ready = {
	5763 = {
		has_province_modifier = verne_port_of_adventure
		has_great_project = {
			type = kazakesh
			tier = 3
		}
	}
	383 = {
		has_great_project = {
			type = aur_kes_akasik
			tier = 3
		}
	}
}


verne_overhaul_akasik_access_ready = {
	akasik_region = {
		type = all
		OR = {
			owned_by = ROOT
			owned_by = A47
			owner = { NOT = { liberty_desire = 10 } }
		}
	}
}

verne_overhaul_in_search_colonizer_route = {
	num_of_colonists = 1
	num_of_explorers = 1
}

verne_overhaul_in_search_subject_projection_route = {
	OR = {
		any_subject_country = { capital_scope = { region = akasik_region } }
		any_ally = {
			OR = {
				capital_scope = { region = akasik_region }
				any_subject_country = {
					num_of_owned_provinces_with = {
						value = 5
						region = akasik_region
					}
				}
				num_of_owned_provinces_with = {
					value = 15
					region = akasik_region
				}
			}
		}
	}
}

verne_overhaul_in_search_network_or_monument_route = {
	OR = {
		verne_overhaul_has_adventure_network_started = yes
		any_owned_province = { has_province_modifier = verne_port_of_adventure }
		383 = {
			has_great_project = {
				type = aur_kes_akasik
				tier = 2
			}
		}
	}
}

# Dynasty safeguard triggers (authoritative)
# Ownership note:
# - These triggers were consolidated from zzz_verne_overhaul_dynasty_triggers.txt.

verne_overhaul_is_verne_trigger = {
	is_or_was_tag = { tag = A33 }
}

verne_overhaul_no_heir_trigger = {
	NOT = { has_heir = yes }
}

verne_overhaul_placeholder_heir_trigger = {
	has_heir = yes
	heir_adm = 0
	heir_dip = 0
	heir_mil = 0
}

verne_overhaul_invalid_heir_state_trigger = {
	OR = {
		verne_overhaul_no_heir_trigger = yes
		verne_overhaul_placeholder_heir_trigger = yes
	}
}

verne_overhaul_should_run_heir_safeguard_trigger = {
	verne_overhaul_is_verne_trigger = yes
	verne_overhaul_invalid_heir_state_trigger = yes
}
